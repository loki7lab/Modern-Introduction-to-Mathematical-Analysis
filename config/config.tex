% -------------------------- 1. 路径与外部配置 --------------------------

    \newcommand{\basepath}{./}
\fi

% -------------------------- 2. 页面尺寸与边距 --------------------------
\geometry{ 
    papersize={210mm,285mm}, % 页面尺寸（接近A4，略短）
    left=22mm, right=22mm,   % 左右边距
    top=28mm, bottom=28mm,   % 上下边距
    headheight=14pt, headsep=20pt % 页眉高度与间距
}


% -------------------------- 3. 章节标题样式 --------------------------
% 部分（Part）标题：新页+居中+特大号加粗
\titleformat{\part}{\newpage\centering\Huge\bfseries}{}{0em}{}
\titlespacing*{\part}{10pt}{10ex}{3.0ex} % 部分标题间距

% 章节（Chapter）标题：仿原书“第一章  实  数”
\newcommand{\chaptertitleformat}[1]{%
    \makebox[4em][s]{#1}%
}

\titleformat{\chapter}[block]
    {\thispagestyle{empty}\centering\zihao{3}\bfseries}
    {第\chinese{chapter}章}
    {2.5em}
    {\chaptertitleformat}

\titlespacing*{\chapter}
    {0pt}
    {0.15\textheight}
    {2.0em}

% 小节（Section）标题：大号加粗+带§符号
\titleformat{\section}{\centering\large\bfseries}{§\arabic{section}}{0.8em}{}
\titlespacing*{\section}{0pt}{2.5ex}{1.5ex} % 小节标题间距

% 自定义未编号章节（Explanation）：封装chapter*，复用其样式
\newcommand{\explanation}[1]{\chapter*{#1}}


% 部分（Part）标题：仿原书标题页
\newcommand{\bookpart}[2]{%
    \cleardoublepage
    \phantomsection
    \addcontentsline{toc}{part}{#1\quad #2}%
    \thispagestyle{empty}%
    \vspace*{0.24\textheight}%
    \begin{center}
        {\zihao{2}\bfseries\makebox[5em][s]{#1}}\par
        \vspace{2.2em}
        {\zihao{3}\bfseries\makebox[7em][s]{#2}}\par
    \end{center}
    \clearpage
}

% -------------------------- 目录格式 --------------------------

% 目录标题：“目  录”居中
\renewcommand{\contentsname}{\makebox[4em][s]{目录}}
\renewcommand{\cfttoctitlefont}{\hfill\zihao{3}\bfseries}
\renewcommand{\cftaftertoctitle}{\hfill}
\setlength{\cftbeforetoctitleskip}{0.14\textheight}
\setlength{\cftaftertoctitleskip}{2.5em}

% 目录只显示到 section
\setcounter{tocdepth}{1}

% section 编号只显示本章内编号：1, 2, 3...
% 目录中再加 §，得到 “§ 1”
\renewcommand{\thesection}{\arabic{section}}

% hyperref 锚点仍保留 chapter.section，避免不同章的 §1 重名
\renewcommand{\theHchapter}{\arabic{chapter}}
\renewcommand{\theHsection}{\arabic{chapter}.\arabic{section}}

% part 在目录中居中显示，不显示页码
\makeatletter
\renewcommand*{\l@part}[2]{%
    \vskip 1.8em
    \begingroup
        \centering
        \zihao{4}\bfseries #1\par
    \endgroup
    \vskip 1.2em
}
\makeatother

% chapter 目录项：第一章  实数 ........ (21)
\renewcommand{\cftchapfont}{\bfseries}
\renewcommand{\cftchappagefont}{\bfseries(}
\renewcommand{\cftchapafterpnum}{)}

\setlength{\cftchapnumwidth}{4.5em}
\renewcommand{\cftchapleader}{\cftdotfill{\cftdotsep}}

% section 目录项：§ 1  实数的无尽小数表示与顺序 ........ (21)
\renewcommand{\cftsecfont}{\normalfont}
\renewcommand{\cftsecpagefont}{\normalfont(}
\renewcommand{\cftsecafterpnum}{)}
\renewcommand{\cftsecpresnum}{\S\ }
\renewcommand{\cftsecaftersnum}{\quad}
\setlength{\cftsecindent}{2em}
\setlength{\cftsecnumwidth}{3em}
\renewcommand{\cftsecleader}{\cftdotfill{\cftdotsep}}

% 条目间距
\setlength{\cftbeforechapskip}{0.8em}
\setlength{\cftbeforesecskip}{0.35em}

% -------------------------- 4. 段落与列表格式 --------------------------
\parindent=2em % 段落首行缩进2字符

% 有序/无序/描述列表统一格式：调整间距
\setenumerate[1]{itemsep=5pt, partopsep=0pt, parsep=\parskip, topsep=5pt}
\setitemize[1]{itemsep=5pt, partopsep=0pt, parsep=\parskip, topsep=5pt}
\setdescription{itemsep=5pt, partopsep=0pt, parsep=\parskip, topsep=5pt}
